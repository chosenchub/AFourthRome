prepare_arabia = {
	enable = {
		original_tag = ENG
		focus_progress = {
			focus = ENG_dip_arabia
			progress > 0.01
		}
		country_exists = ARB
		NOT = { 
			has_completed_focus = ENG_dip_arabia 
		}
	}
	
	abort = {
		OR = {
			NOT = { country_exists = ARB }
			AND = {
				has_completed_focus = ENG_dip_arabia
				ENG = { has_war_with = ARB }
			}
		}
	}
	
	ai_strategy = {
		type = conquer
		id = "ARB"
		value = 100
	}
}
