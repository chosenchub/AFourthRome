# Death of Sultan Abdulmecid II
country_event = {
	id = tur.2
	title = tur.2.t
	desc = tur.2.d
	picture = GFX_report_event_europe_funeral

	fire_only_once = yes

	trigger = {
		tag = TUR
		date > 1936.1.22
	}

	mean_time_to_happen = { days = 2 }

	option = {
		name = tur.2.a

		custom_effect_tooltip = TUR_emergency_session_focus_unlocked_tt

		kill_country_leader = yes

		set_country_flag = TUR_assassination_of_the_sultan

		if = {
			limit = {
				TUR = { 
					NOT = { 
						has_idea = TUR_political_turmoil 
					}
				}
			}
			add_ideas = TUR_political_turmoil
		}

		create_country_leader = {
			name = "The General Assembly"
			desc = "POLITICS_GENERAL_ASSEMBLY_DESC"
			picture = "Portrait_Turkey_General_Assembly.tga"
			expire = "1965.1.1"
			ideology = social_liberal_subtype
			traits = {}
		}

		hidden_effect = {
			create_country_leader = {
				name = "The General Assembly"
				desc = "POLITICS_GENERAL_ASSEMBLY_DESC"
				picture = "Portrait_Turkey_General_Assembly.tga"
				expire = "1965.1.1"
				ideology = paternal_autocrat_subtype
				traits = {}
			}
			
			random_country = {
				limit = { 
					NOT = {
						OR = {
							tag = TUR
							original_tag = TUR
						} 
					} 
				}
				hidden_effect = { 
					news_event = { id = worldnews.1 days = 4 } 
				}
			}
		}

		add_named_threat = { threat = 2 name = TUR_assassination_of_the_sultan_threat }
	}
}
